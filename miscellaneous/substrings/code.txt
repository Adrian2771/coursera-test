<script>
	s = 'azcbobobegghakl';
	var temp = s[0];

	var j = 0;
	while(j <= s.length){
		var i = j + 1;
		while(i <= s.length){
			//console.log(s.substring(j, i));
			s1 = s.substring(j, i);
			var k = 0;
			while(k < s1.length - 1){
				if(s1.charCodeAt(k) <= s1.charCodeAt(k + 1)){
					if(temp.length < (s1.substring(0, k + 2)).length){
						temp = s1.substring(0, k + 2);
					}
					k++;
				}
				else{
					break;
				}
			}
			i++;
		}
		j++;
	}
	console.log("The longest substring in alphabetical order is: " + temp);
</script>
